142 public DamageDetails TakeDamage(Move move, Pokemon attacker)
143 {
144 float critical = 1;
145
146 if (Random.value * 100f <= 6.25) critical = 2;
147
148 float effectiveness = TypeChart.GetEffectiveness(move.Base.Type, Base.Type1) * TypeChart.GetEffectiveness(move.Base.Type, Base.Type2);
149
150 DamageDetails damageDetails = new()
151 {
152 TypeEffectiveness = effectiveness,
153 Critical = critical,
154 Fainted = false
155 };
156
157 bool specialMove = move.Base.Category == MoveCategory.Special;
158
159 float attack = specialMove ? attacker.SpecialAttack : attacker.Attack;
160 float defense = specialMove ? attacker.SpecialDefense : attacker.Defense;
161
162 float modifiers = Random.Range(0.85f, 1f) * effectiveness * critical;
163 float a = ((2 * attacker.Level) + 10) / 250f;
164
165 float d = (a * move.Base.Power * ((float)attack / defense)) + 2;
166 int damage = Mathf.FloorToInt(d * modifiers);
167
168 DecreaseHP(damage);
169
170 return damageDetails;
171 }

355 private void CalculateStats()
356 {
357 Stats = new Dictionary<Stat, int>
358 {
359 { Stat.Attack, Mathf.FloorToInt(Base.Attack * Level / 100f) + 5 },
360 { Stat.Defense, Mathf.FloorToInt(Base.Defense * Level / 100f) + 5 },
361 { Stat.SpecialAttack, Mathf.FloorToInt(Base.SpecialAttack * Level / 100f) + 5 },
362 { Stat.SpecialDefense, Mathf.FloorToInt(Base.SpecialDefense * Level / 100f) + 5 },
363 { Stat.Speed, Mathf.FloorToInt(Base.Speed * Level / 100f) + 5 }
364 };
365
366 MaxHp = Mathf.FloorToInt(Base.MaxHp * Level / 100f) + 10 + Level;
367 }

369 private int GetStat(Stat stat)
370 {
371 int statVal = Stats[stat];
372
373 int boost = StatBoosts[stat];
374 float[] boostValues = new float[] { 1f, 1.5f, 2f, 2.5f, 3f, 3.5f, 4f };
375
376 if (boost >= 0)
377 statVal = Mathf.FloorToInt(statVal * boostValues[boost]);
378 else
379 statVal = Mathf.FloorToInt(statVal / boostValues[-boost]);
380
381 return statVal;
382 }
