498 private bool CheckIfMoveHits(Move move, Pokemon source, Pokemon target)
499 {
500 if (move.Base.AlwaysHits) return true;
501
502 float moveAccuracy = move.Base.Accuracy;
503
504 int accuracy = source.StatBoosts[Stat.Accuracy];
505 int evasion = target.StatBoosts[Stat.Evasion];
506
507 float[] boostValues = new float[] { 1f, 4f / 3f, 5f / 3f, 2f, 7f / 3f, 8f / 3f, 3f };
508
509 if (accuracy > 0)
510 moveAccuracy *= boostValues[accuracy];
511 else
512 moveAccuracy /= boostValues[-accuracy];
513
514 if (evasion > 0)
515 moveAccuracy /= boostValues[evasion];
516 else
517 moveAccuracy *= boostValues[-evasion];
518
519 return UnityEngine.Random.Range(1, 101) <= moveAccuracy;
520 }

550 private IEnumerator RunMove(BattleUnit source, BattleUnit target, Move move)
551 {
552 bool canRunMove = source.Pokemon.OnBeforeTurn();
553 if (!canRunMove)
554 {
555 yield return ShowStatusChanges(source.Pokemon);
556 yield return source.Hud.WaitForHPUpdate();
557 yield break;
558 }
559
560 yield return ShowStatusChanges(source.Pokemon);
561
562 move.PP--;
563
564 yield return dialogBox.TypeDialog($"{source.Pokemon.Base.Name} used {move.Base.MoveName}");
565
566 if (CheckIfMoveHits(move, source.Pokemon, target.Pokemon))
567 {
568 source.PlayAttackAnimation();
569 yield return new WaitForSeconds(1f);
570
571 target.PlayHitAnimation();
572
573 if (move.Base.Category == MoveCategory.Status)
574 {
575 yield return RunMoveEffects(move.Base.Effects, source.Pokemon, target.Pokemon, move.Base.Target);
576 }
577 else
578 {
579 DamageDetails damageDetails = target.Pokemon.TakeDamage(move, source.Pokemon);
580 yield return target.Hud.WaitForHPUpdate();
581 yield return ShowDamageDetails(damageDetails);
582 }
583
584 if (target.Pokemon.HP <= 0)
585 {
586 yield return HandlePokemonFainted(target);
587 }
588 }
589 else
590 {
591 yield return dialogBox.TypeDialog($"{source.Pokemon.Base.Name}'s attack missed!");
592 }
593
594 if (move.Base.SecondaryEffects != null && move.Base.SecondaryEffects.Count > 0 && target.Pokemon.HP > 0)
595 {
596 foreach (SecondaryEffects secondaryEffect in move.Base.SecondaryEffects)
597 {
598 int random = UnityEngine.Random.Range(1, 101);
599
600 if (random <= secondaryEffect.Chance)
601 {
602 yield return RunMoveEffects(secondaryEffect, source.Pokemon, target.Pokemon, secondaryEffect.Target);
603 }
604 }
605 }

608 private IEnumerator RunAfterTurn(BattleUnit source)
609 {
610 if (state == BattleState.BattleOver) yield break;
611 yield return new WaitUntil(() => state == BattleState.RunningTurn);
612
613 source.Pokemon.OnAfterTurn();
614 yield return ShowStatusChanges(source.Pokemon);
615 yield return source.Hud.WaitForHPUpdate();
616
617 // Since statuses like poison and burn effect the HP of the pokemon
618 // it has the potential to faint because of the effect
619 // so this if statement checks if that is the case and handles it accordingly
620 if (source.Pokemon.HP <= 0)
621 {
622 yield return HandlePokemonFainted(source);
623 yield return new WaitUntil(() => state == BattleState.RunningTurn);
624 }
625 }

627 private IEnumerator RunMoveEffects(MoveEffects effects, Pokemon source, Pokemon target, MoveTarget moveTarget)
628 {
629 // Stat boosting/deboosting
630 if (effects.Boosts != null)
631 {
632 if (moveTarget == MoveTarget.self)
633 source.ApplyBoosts(effects.Boosts);
634 else
635 target.ApplyBoosts(effects.Boosts);
636 }
637
638 // Status Conditions/Effects
639 if (effects.Status != ConditionID.none)
640 {
641 target.SetStatus(effects.Status);
642 }
643
644 // Volatile Status Conditions/Effects
645 if (effects.VolatileStatus != ConditionID.none)
646 {
647 target.SetVolatileStatus(effects.VolatileStatus);
648 }
649
650 yield return ShowStatusChanges(source);
651 yield return ShowStatusChanges(target);
652 }

654 private IEnumerator RunTurns(BattleAction playerAction)
655 {
656 state = BattleState.RunningTurn;
657
658 if (playerAction == BattleAction.Move)
659 {
660 playerUnit.Pokemon.CurrentMove = playerUnit.Pokemon.Moves[currentMoveIndex];
661 enemyUnit.Pokemon.CurrentMove = enemyUnit.Pokemon.GetRandomMove();
662
663 int playerMovePriority = playerUnit.Pokemon.CurrentMove.Base.Priority;
664 int enemyMovePriority = enemyUnit.Pokemon.CurrentMove.Base.Priority;
665
666 // Check to see who goes first
667
668 bool playerGoesFirst = true;
669
670 if (enemyMovePriority > playerMovePriority)
671 playerGoesFirst = false;
672 else if (enemyMovePriority == playerMovePriority)
673 playerGoesFirst = playerUnit.Pokemon.Speed >= enemyUnit.Pokemon.Speed;
674
675 BattleUnit firstUnit = playerGoesFirst ? playerUnit : enemyUnit;
676 BattleUnit secondUnit = playerGoesFirst ? enemyUnit : playerUnit;
677
678 Pokemon secondPokemon = secondUnit.Pokemon;
679
680 // First turn
681 yield return RunMove(firstUnit, secondUnit, firstUnit.Pokemon.CurrentMove);
682 yield return RunAfterTurn(firstUnit);
683 if (state == BattleState.BattleOver) yield break;
684
685 if (secondPokemon.HP > 0)
686 {
687 // Second turn
688 yield return RunMove(secondUnit, firstUnit, secondUnit.Pokemon.CurrentMove);
689 yield return RunAfterTurn(secondUnit);
690 if (state == BattleState.BattleOver) yield break;
691 }
692 }
693 else
694 {
695 switch (playerAction)
696 {
697 case BattleAction.SwitchPokemon:
698 Pokemon selectedPokemon = partyScreen.SelectedMember;
699 state = BattleState.Busy;
700 yield return SwitchPokemon(selectedPokemon);
701 break;
702 case BattleAction.UseItem:
703 // This is handled from the item screen, so do nothing and skip to the enemy's turn
704 dialogBox.SetActionSelectorActive(false);
705 break;
706 case BattleAction.Run:
707 yield return TryToEscape();
708 break;
709 }
710
711 // Enemy move
712 Move enemyMove = enemyUnit.Pokemon.GetRandomMove();
713
714 yield return RunMove(enemyUnit, playerUnit, enemyMove);
715 yield return RunAfterTurn(enemyUnit);
716
717 if (state == BattleState.BattleOver) yield break;
718 }
719
720 if (state != BattleState.BattleOver) ActionSelection();
721 }

727 private IEnumerator HandlePokemonFainted(BattleUnit faintedUnit)
728 {
729 yield return dialogBox.TypeDialog($"{faintedUnit.Pokemon.Base.Name} Fainted");
730 faintedUnit.PlayFaintAnimation();
731
732 yield return new WaitForSeconds(1.5f);
733
734 if (!faintedUnit.IsPlayerUnit)
735 {
736 int expYield = faintedUnit.Pokemon.Base.ExpYield;
737 int enemyLevel = faintedUnit.Pokemon.Level;
738 float trainerBonus = isTrainerBattle ? 1.5f : 1f;
739
740 int expGain = Mathf.FloorToInt(expYield * enemyLevel * trainerBonus / 7);
741 playerUnit.Pokemon.Exp += expGain;
742
743 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} gained {expGain} exp!");
744 yield return playerUnit.Hud.SetExpSmooth();
745
746 yield return new WaitForSeconds(1f);
747
748 // "while" not "if" to handle multiple level ups at once
749 while (playerUnit.Pokemon.CheckForLevelUp())
750 {
751 playerUnit.Hud.SetLevel();
752 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} grew to level {playerUnit.Pokemon.Level}!");
753
754 // Try to learn new move
755 LearnableMove newMove = playerUnit.Pokemon.GetLearnableMoveAtCurrentLevel();
756
757 if (newMove != null)
758 {
759 if (playerUnit.Pokemon.Moves.Count < PokemonBase.MaxNumOfMoves)
760 {
761 playerUnit.Pokemon.LearnMove(newMove);
762 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} learned {newMove.MoveBase.MoveName}!");
763 dialogBox.SetMoveNames(playerUnit.Pokemon.Moves);
764 }
765 else
766 {
767 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} is trying to learn {newMove.MoveBase.MoveName}");
768 yield return dialogBox.TypeDialog($"But it cannot learn more than {PokemonBase.MaxNumOfMoves} moves!");
769 yield return ChooseMoveToForget(playerUnit.Pokemon, newMove.MoveBase);
770 yield return new WaitUntil(() => state != BattleState.MoveToForget);
771 yield return new WaitForSeconds(2f);
772 }
773 }
774
775 yield return playerUnit.Hud.SetExpSmooth(true);
776 }
777
778 yield return new WaitForSeconds(1f);
779 }
780
781 CheckForBattleOver(faintedUnit);
782 }

784 private IEnumerator ThrowPokeball(PokeballItem pokeballItem)
785 {
786 state = BattleState.Busy;
787
788 if (isTrainerBattle)
789 {
790
791 yield return dialogBox.TypeDialog("You can't catch the trainer's pokemon!");
792 state = BattleState.RunningTurn;
793 yield break;
794 }
795
796 yield return dialogBox.TypeDialog($"{player.Name} used {pokeballItem.Name}!");
797
798 GameObject pokeballObject = Instantiate(pokeball, playerUnit.transform.position - new Vector3(2, 0), Quaternion.identity);
799
800 SpriteRenderer pokeballSprite = pokeballObject.GetComponent<SpriteRenderer>();
801 pokeballSprite.sprite = pokeballItem.Icon;
802
803 // Animations
804
805 yield return pokeballSprite.transform.DOJump(enemyUnit.transform.position + new Vector3(0, 2), 2f, 1, 1f).WaitForCompletion();
806 yield return enemyUnit.PlayCaptureAnimation();
807 yield return pokeballSprite.transform.DOMoveY(enemyUnit.transform.position.y - 1.3f, 0.5f).WaitForCompletion();
808
809 // Shake
810 int shakeCount = GetShakeCount(enemyUnit.Pokemon, pokeballItem);
811
812 for (int i = 0; i < Mathf.Min(shakeCount, 3f); ++i)
813 {
814 yield return new WaitForSeconds(0.5f);
815 yield return pokeballSprite.transform.DOPunchRotation(new Vector3(0f, 0f, 10f), 0.8f).WaitForCompletion();
816 }
817
818 if (shakeCount == 4)
819 {
820 // Pokemon caught
821 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} was caught!");
822 yield return new WaitForSeconds(0.5f);
823
824 yield return pokeballSprite.DOFade(0f, 1.5f).WaitForCompletion();
825 Destroy(pokeballObject);
826
827 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} has been added to your party!");
828 yield return new WaitForSeconds(0.5f);
829
830 playerParty.AddPokemon(enemyUnit.Pokemon);
831 BattleOver(true);
832 }
833 else
834 {
835 // Pokemon broke out
836 yield return new WaitForSeconds(1f);
837 pokeballSprite.DOFade(0f, 0.2f);
838 yield return enemyUnit.PlayBreakoutAnimation();
839
840 if (shakeCount < 2)
841 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} broke free!");
842 else
843 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} was almost caught!");
844
845 Destroy(pokeballObject);
846 state = BattleState.RunningTurn;
847 }
848 }

850 private IEnumerator SwitchPokemon(Pokemon newPokemon, bool isTrainerAboutToUse = false)
851 {
852 if (playerUnit.Pokemon.HP > 0)
853 {
854 yield return dialogBox.TypeDialog($"Come back {playerUnit.Pokemon.Base.name}");
855 playerUnit.PlayFaintAnimation();
856 yield return new WaitForSeconds(1.5f);

886 private IEnumerator TryToEscape()
887 {
888 state = BattleState.Busy;
889 escapeAttempts++;
890
891 if (isTrainerBattle)
892 {
893 yield return dialogBox.TypeDialog("You can't run from trainer battles!");
894 state = BattleState.RunningTurn;
895 yield break;
896 }
897
898 int playerSpeed = playerUnit.Pokemon.Speed;
899 int enemySpeed = enemyUnit.Pokemon.Speed;
900
901 if (enemySpeed < playerSpeed)
902 {
903 yield return dialogBox.TypeDialog("Safely ran away!");
904
905 playerUnit.PlayFaintAnimation();
906 yield return new WaitForSeconds(1.5f);
907
908 BattleOver(true);
909 }
910 else
911 {
912 int f = ((playerSpeed * 128 / enemySpeed) + 30) * escapeAttempts;
913 f %= 256;
914
915 if (UnityEngine.Random.Range(0, 256) < f)
916 {
917 yield return dialogBox.TypeDialog("Safely ran away!");
918 playerUnit.PlayFaintAnimation();
919 BattleOver(true);
920 }
921 else
922 {
923 yield return dialogBox.TypeDialog("Failed to escape!");
924 state = BattleState.RunningTurn;
925 }

929 private int GetShakeCount(Pokemon pokemon, PokeballItem pokeballItem)
930 {
931 float a = ((3 * pokemon.MaxHp) - (2 * pokemon.HP)) * pokemon.Base.CatchRate * pokeballItem.CatchRateModifier * ConditionsDB.GetStatusBonus(pokemon.Status) / (3 * pokemon.MaxHp);
932
933 if (a >= 255)
934 // Pokemon caught
935 return 4;
936
937 float b = 1048560 / Mathf.Sqrt(Mathf.Sqrt(16711680 / a));
938
939 int shakeCount = 0;
940
941 while (shakeCount < 4)
942 {
943 if (UnityEngine.Random.Range(0, 65535) >= b) break;
944
945 ++shakeCount;
946 }
947
948 return shakeCount;
