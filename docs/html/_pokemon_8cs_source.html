114 public DamageDetails TakeDamage(Move move, Pokemon attacker)
115 {
116 float critical = 1;
117
118 if (Random.value * 100f <= 6.25) critical = 2;
119
120 float effectiveness = TypeChart.GetEffectiveness(move.Base.Type, Base.Type1) * TypeChart.GetEffectiveness(move.Base.Type, Base.Type2);
121
122 DamageDetails damageDetails = new()
123 {
124 TypeEffectiveness = effectiveness,
125 Critical = critical,
126 Fainted = false
127 };
128
129 bool specialMove = move.Base.Category == MoveCategory.Special;
130
131 float attack = specialMove ? attacker.SpecialAttack : attacker.Attack;
132 float defense = specialMove ? attacker.SpecialDefense : attacker.Defense;
133
134 float modifiers = Random.Range(0.85f, 1f) * effectiveness * critical;
135 float a = ((2 * attacker.Level) + 10) / 250f;
136
137 float d = (a * move.Base.Power * ((float)attack / defense)) + 2;
138 int damage = Mathf.FloorToInt(d * modifiers);
139
140 DecreaseHP(damage);
141
142 return damageDetails;
143 }

327 private void CalculateStats()
328 {
329 Stats = new Dictionary<Stat, int>
330 {
331 { Stat.Attack, Mathf.FloorToInt(Base.Attack * Level / 100f) + 5 },
332 { Stat.Defense, Mathf.FloorToInt(Base.Defense * Level / 100f) + 5 },
333 { Stat.SpecialAttack, Mathf.FloorToInt(Base.SpecialAttack * Level / 100f) + 5 },
334 { Stat.SpecialDefense, Mathf.FloorToInt(Base.SpecialDefense * Level / 100f) + 5 },
335 { Stat.Speed, Mathf.FloorToInt(Base.Speed * Level / 100f) + 5 }
336 };
337
338 MaxHp = Mathf.FloorToInt(Base.MaxHp * Level / 100f) + 10 + Level;
339 }

341 private int GetStat(Stat stat)
342 {
343 int statVal = Stats[stat];
344
345 int boost = StatBoosts[stat];
346 float[] boostValues = new float[] { 1f, 1.5f, 2f, 2.5f, 3f, 3.5f, 4f };
347
348 if (boost >= 0)
349 statVal = Mathf.FloorToInt(statVal * boostValues[boost]);
350 else
351 statVal = Mathf.FloorToInt(statVal / boostValues[-boost]);
352
353 return statVal;
354 }
