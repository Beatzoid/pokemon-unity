476 private bool CheckIfMoveHits(Move move, Pokemon source, Pokemon target)
477 {
478 if (move.Base.AlwaysHits) return true;
479
480 float moveAccuracy = move.Base.Accuracy;
481
482 int accuracy = source.StatBoosts[Stat.Accuracy];
483 int evasion = target.StatBoosts[Stat.Evasion];
484
485 float[] boostValues = new float[] { 1f, 4f / 3f, 5f / 3f, 2f, 7f / 3f, 8f / 3f, 3f };
486
487 if (accuracy > 0)
488 moveAccuracy *= boostValues[accuracy];
489 else
490 moveAccuracy /= boostValues[-accuracy];
491
492 if (evasion > 0)
493 moveAccuracy /= boostValues[evasion];
494 else
495 moveAccuracy *= boostValues[-evasion];
496
497 return UnityEngine.Random.Range(1, 101) <= moveAccuracy;
498 }

528 private IEnumerator RunMove(BattleUnit source, BattleUnit target, Move move)
529 {
530 bool canRunMove = source.Pokemon.OnBeforeTurn();
531 if (!canRunMove)
532 {
533 yield return ShowStatusChanges(source.Pokemon);
534 yield return source.Hud.WaitForHPUpdate();
535 yield break;
536 }
537
538 yield return ShowStatusChanges(source.Pokemon);
539
540 move.PP--;
541
542 yield return dialogBox.TypeDialog($"{source.Pokemon.Base.Name} used {move.Base.MoveName}");
543
544 if (CheckIfMoveHits(move, source.Pokemon, target.Pokemon))
545 {
546 source.PlayAttackAnimation();
547 yield return new WaitForSeconds(1f);
548
549 target.PlayHitAnimation();
550
551 if (move.Base.Category == MoveCategory.Status)
552 {
553 yield return RunMoveEffects(move.Base.Effects, source.Pokemon, target.Pokemon, move.Base.Target);
554 }
555 else
556 {
557 DamageDetails damageDetails = target.Pokemon.TakeDamage(move, source.Pokemon);
558 yield return target.Hud.WaitForHPUpdate();
559 yield return ShowDamageDetails(damageDetails);
560 }
561
562 if (target.Pokemon.HP <= 0)
563 {
564 yield return HandlePokemonFainted(target);
565 }
566 }
567 else
568 {
569 yield return dialogBox.TypeDialog($"{source.Pokemon.Base.Name}'s attack missed!");
570 }
571
572 if (move.Base.SecondaryEffects != null && move.Base.SecondaryEffects.Count > 0 && target.Pokemon.HP > 0)
573 {
574 foreach (SecondaryEffects secondaryEffect in move.Base.SecondaryEffects)
575 {
576 int random = UnityEngine.Random.Range(1, 101);
577
578 if (random <= secondaryEffect.Chance)
579 {
580 yield return RunMoveEffects(secondaryEffect, source.Pokemon, target.Pokemon, secondaryEffect.Target);
581 }
582 }
583 }
584 }

586 private IEnumerator RunAfterTurn(BattleUnit source)
587 {
588 if (state == BattleState.BattleOver) yield break;
589 yield return new WaitUntil(() => state == BattleState.RunningTurn);
590
591 source.Pokemon.OnAfterTurn();
592 yield return ShowStatusChanges(source.Pokemon);
593 yield return source.Hud.WaitForHPUpdate();
594
595 // Since statuses like poison and burn effect the HP of the pokemon
596 // it has the potential to faint because of the effect
597 // so this if statement checks if that is the case and handles it accordingly
598 if (source.Pokemon.HP <= 0)
599 {
600 yield return HandlePokemonFainted(source);
601 yield return new WaitUntil(() => state == BattleState.RunningTurn);
602 }
603 }

605 private IEnumerator RunMoveEffects(MoveEffects effects, Pokemon source, Pokemon target, MoveTarget moveTarget)
606 {
607 // Stat boosting/deboosting
608 if (effects.Boosts != null)
609 {
610 if (moveTarget == MoveTarget.self)
611 source.ApplyBoosts(effects.Boosts);
612 else
613 target.ApplyBoosts(effects.Boosts);
614 }
615
616 // Status Conditions/Effects
617 if (effects.Status != ConditionID.none)
618 {
619 target.SetStatus(effects.Status);
620 }
621
622 // Volatile Status Conditions/Effects
623 if (effects.VolatileStatus != ConditionID.none)
624 {
625 target.SetVolatileStatus(effects.VolatileStatus);
626 }
627
628 yield return ShowStatusChanges(source);
629 yield return ShowStatusChanges(target);
630 }

632 private IEnumerator RunTurns(BattleAction playerAction)
633 {
634 state = BattleState.RunningTurn;
635
636 if (playerAction == BattleAction.Move)
637 {
638 playerUnit.Pokemon.CurrentMove = playerUnit.Pokemon.Moves[currentMoveIndex];
639 enemyUnit.Pokemon.CurrentMove = enemyUnit.Pokemon.GetRandomMove();
640
641 int playerMovePriority = playerUnit.Pokemon.CurrentMove.Base.Priority;
642 int enemyMovePriority = enemyUnit.Pokemon.CurrentMove.Base.Priority;
643
644 // Check to see who goes first
645
646 bool playerGoesFirst = true;
647
648 if (enemyMovePriority > playerMovePriority)
649 playerGoesFirst = false;
650 else if (enemyMovePriority == playerMovePriority)
651 playerGoesFirst = playerUnit.Pokemon.Speed >= enemyUnit.Pokemon.Speed;
652
653 BattleUnit firstUnit = playerGoesFirst ? playerUnit : enemyUnit;
654 BattleUnit secondUnit = playerGoesFirst ? enemyUnit : playerUnit;
655
656 Pokemon secondPokemon = secondUnit.Pokemon;
657
658 // First turn
659 yield return RunMove(firstUnit, secondUnit, firstUnit.Pokemon.CurrentMove);
660 yield return RunAfterTurn(firstUnit);
661 if (state == BattleState.BattleOver) yield break;
662
663 if (secondPokemon.HP > 0)
664 {
665 // Second turn
666 yield return RunMove(secondUnit, firstUnit, secondUnit.Pokemon.CurrentMove);
667 yield return RunAfterTurn(secondUnit);
668 if (state == BattleState.BattleOver) yield break;
669 }
670 }
671 else
672 {
673 switch (playerAction)
674 {
675 case BattleAction.SwitchPokemon:
676 Pokemon selectedPokemon = partyScreen.SelectedMember;
677 state = BattleState.Busy;
678 yield return SwitchPokemon(selectedPokemon);
679 break;
680 case BattleAction.UseItem:
681 // This is handled from the item screen, so do nothing and skip to the enemy's turn
682 dialogBox.SetActionSelectorActive(false);
683 break;
684 case BattleAction.Run:
685 yield return TryToEscape();
686 break;
687 }
688
689 // Enemy move
690 Move enemyMove = enemyUnit.Pokemon.GetRandomMove();
691
692 yield return RunMove(enemyUnit, playerUnit, enemyMove);
693 yield return RunAfterTurn(enemyUnit);
694
695 if (state == BattleState.BattleOver) yield break;
696 }
697
698 if (state != BattleState.BattleOver) ActionSelection();
699 }

705 private IEnumerator HandlePokemonFainted(BattleUnit faintedUnit)
706 {
707 yield return dialogBox.TypeDialog($"{faintedUnit.Pokemon.Base.Name} Fainted");
708 faintedUnit.PlayFaintAnimation();
709
710 yield return new WaitForSeconds(1.5f);
711
712 if (!faintedUnit.IsPlayerUnit)
713 {
714 int expYield = faintedUnit.Pokemon.Base.ExpYield;
715 int enemyLevel = faintedUnit.Pokemon.Level;
716 float trainerBonus = isTrainerBattle ? 1.5f : 1f;
717
718 int expGain = Mathf.FloorToInt(expYield * enemyLevel * trainerBonus / 7);
719 playerUnit.Pokemon.Exp += expGain;
720
721 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} gained {expGain} exp!");
722 yield return playerUnit.Hud.SetExpSmooth();
723
724 yield return new WaitForSeconds(1f);
725
726 // "while" not "if" to handle multiple level ups at once
727 while (playerUnit.Pokemon.CheckForLevelUp())
728 {
729 playerUnit.Hud.SetLevel();
730 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} grew to level {playerUnit.Pokemon.Level}!");
731
732 // Try to learn new move
733 LearnableMove newMove = playerUnit.Pokemon.GetLearnableMoveAtCurrentLevel();
734
735 if (newMove != null)
736 {
737 if (playerUnit.Pokemon.Moves.Count < PokemonBase.MaxNumOfMoves)
738 {
739 playerUnit.Pokemon.LearnMove(newMove);
740 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} learned {newMove.MoveBase.MoveName}!");
741 dialogBox.SetMoveNames(playerUnit.Pokemon.Moves);
742 }
743 else
744 {
745 yield return dialogBox.TypeDialog($"{playerUnit.Pokemon.Base.Name} is trying to learn {newMove.MoveBase.MoveName}");
746 yield return dialogBox.TypeDialog($"But it cannot learn more than {PokemonBase.MaxNumOfMoves} moves!");
747 yield return ChooseMoveToForget(playerUnit.Pokemon, newMove.MoveBase);
748 yield return new WaitUntil(() => state != BattleState.MoveToForget);
749 yield return new WaitForSeconds(2f);
750 }
751 }
752
753 yield return playerUnit.Hud.SetExpSmooth(true);
754 }
755
756 yield return new WaitForSeconds(1f);
757 }
758
759 CheckForBattleOver(faintedUnit);
760 }
761 private IEnumerator ThrowPokeball()
762 {
763 state = BattleState.Busy;
764
765 if (isTrainerBattle)
766 {
767
768 yield return dialogBox.TypeDialog("You can't catch the trainer's pokemon!");
769 state = BattleState.RunningTurn;
770 yield break;
771 }
772
773 yield return dialogBox.TypeDialog($"{player.Name} used Pokeball!");
774
775 GameObject pokeballObject = Instantiate(pokeball, playerUnit.transform.position - new Vector3(2, 0), Quaternion.identity);
776 SpriteRenderer pokeballSprite = pokeballObject.GetComponent<SpriteRenderer>();
777
778 // Animations
779
780 yield return pokeballSprite.transform.DOJump(enemyUnit.transform.position + new Vector3(0, 2), 2f, 1, 1f).WaitForCompletion();
781 yield return enemyUnit.PlayCaptureAnimation();
782 yield return pokeballSprite.transform.DOMoveY(enemyUnit.transform.position.y - 1.3f, 0.5f).WaitForCompletion();
783
784 // Shake
785 int shakeCount = GetShakeCount(enemyUnit.Pokemon);
786
787 for (int i = 0; i < Mathf.Min(shakeCount, 3f); ++i)
788 {
789 yield return new WaitForSeconds(0.5f);
790 yield return pokeballSprite.transform.DOPunchRotation(new Vector3(0f, 0f, 10f), 0.8f).WaitForCompletion();
791 }
792
793 if (shakeCount == 4)
794 {
795 // Pokemon caught
796 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} was caught!");
797 yield return new WaitForSeconds(0.5f);
798
799 yield return pokeballSprite.DOFade(0f, 1.5f).WaitForCompletion();
800 Destroy(pokeballObject);
801
802 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} has been added to your party!");
803 yield return new WaitForSeconds(0.5f);
804
805 playerParty.AddPokemon(enemyUnit.Pokemon);
806 BattleOver(true);
807 }
808 else
809 {
810 // Pokemon broke out
811 yield return new WaitForSeconds(1f);
812 pokeballSprite.DOFade(0f, 0.2f);
813 yield return enemyUnit.PlayBreakoutAnimation();
814
815 if (shakeCount < 2)
816 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} broke free!");
817 else
818 yield return dialogBox.TypeDialog($"{enemyUnit.Pokemon.Base.Name} was almost caught!");
819
820 Destroy(pokeballObject);
821 state = BattleState.RunningTurn;
822 }
823 }

825 private int GetShakeCount(Pokemon pokemon)
826 {
827 float a = ((3 * pokemon.MaxHp) - (2 * pokemon.HP)) * pokemon.Base.CatchRate * ConditionsDB.GetStatusBonus(pokemon.Status) / (3 * pokemon.MaxHp);
828
829 if (a >= 255)
830 // Pokemon caught
831 return 4;
832
833 float b = 1048560 / Mathf.Sqrt(Mathf.Sqrt(16711680 / a));
834
835 int shakeCount = 0;
836
837 while (shakeCount < 4)
838 {
839 if (UnityEngine.Random.Range(0, 65535) >= b) break;
840
841 ++shakeCount;
842 }
843
844 return shakeCount;
845 }

883 private IEnumerator TryToEscape()
884 {
885 state = BattleState.Busy;
886 escapeAttempts++;
887
888 if (isTrainerBattle)
889 {
890 yield return dialogBox.TypeDialog("You can't run from trainer battles!");
891 state = BattleState.RunningTurn;
892 yield break;
893 }
894
895 int playerSpeed = playerUnit.Pokemon.Speed;
896 int enemySpeed = enemyUnit.Pokemon.Speed;
897
898 if (enemySpeed < playerSpeed)
899 {
900 yield return dialogBox.TypeDialog("Safely ran away!");
901
902 playerUnit.PlayFaintAnimation();
903 yield return new WaitForSeconds(1.5f);
904
905 BattleOver(true);
906 }
907 else
908 {
909 int f = ((playerSpeed * 128 / enemySpeed) + 30) * escapeAttempts;
910 f %= 256;
911
912 if (UnityEngine.Random.Range(0, 256) < f)
913 {
914 yield return dialogBox.TypeDialog("Safely ran away!");
915 playerUnit.PlayFaintAnimation();
916 BattleOver(true);
917 }
918 else
919 {
920 yield return dialogBox.TypeDialog("Failed to escape!");
921 state = BattleState.RunningTurn;
922 }
